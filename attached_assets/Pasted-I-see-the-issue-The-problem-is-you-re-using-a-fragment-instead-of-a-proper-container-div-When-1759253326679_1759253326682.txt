I see the issue. The problem is you're using a fragment `<>` instead of a proper container div. When WorkspaceShell wraps your page, the fragment isn't creating the layout structure you need.

Change **`client/src/pages/ProjectEditPageWithSidebar.tsx`** to use a proper container div:

```tsx
import { useLocation } from 'wouter';
import { ProjectContainer } from '@/components/ProjectContainer';
import Header from '@/components/Header';

interface ProjectEditPageWithSidebarProps {
  params: { id: string };
}

export default function ProjectEditPageWithSidebar({ params }: ProjectEditPageWithSidebarProps) {
  const [, setLocation] = useLocation();

  const handleBack = () => {
    setLocation('/projects');
  };

  const handleSearch = (query: string) => {
    setLocation(`/?search=${encodeURIComponent(query)}`);
  };

  const handleNavigate = (view: string) => {
    switch (view) {
      case 'notebook':
        setLocation('/notebook');
        break;
      case 'projects':
        setLocation('/projects');
        break;
      default:
        setLocation('/');
    }
  };

  const handleCreateNew = () => {
    setLocation('/?create=true');
  };

  return (
    <div className="h-screen flex flex-col bg-background">
      {/* Global Navigation Header */}
      <div className="flex-shrink-0">
        <Header 
          onSearch={handleSearch}
          onNavigate={handleNavigate}
          onCreateNew={handleCreateNew}
        />
      </div>
      
      {/* Main Content Area - takes remaining space */}
      <div className="flex-1 overflow-hidden min-h-0">
        <ProjectContainer 
          projectId={params.id}
          onBack={handleBack}
        />
      </div>
    </div>
  );
}
```

The key changes:
- Replace fragment `<>` with a proper div that creates a flex column layout
- Use `h-screen` for full viewport height
- Wrap Header in `flex-shrink-0` to prevent it from being compressed
- Use `flex-1` and `min-h-0` on the content container for proper flex behavior

This should make the Header visible while keeping tabs intact since we're still not touching WorkspaceShell or WorkspaceLayout.
